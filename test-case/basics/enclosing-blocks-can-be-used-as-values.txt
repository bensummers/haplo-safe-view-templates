
Enclosing blocks can be used as values for other functions

---

<div>
    within(b) {
        each(^{a}) {
            <span> b "-" v </span>
        }
    }
</div>

---

{
    "a": [
        {"b":"X", "v":"0"},
        {"b":"Y", "v":"1"}
    ],
    "b": {
    }
}

---

<div><span>X-0</span><span>Y-1</span></div>


---
NEW TEMPLATE
---
<div>
    within(b) {
        if(^{a}) {
            "TRUE"
        } else {
            "FALSE"
        }
    }
</div>


---
{
    "a": true,
    "b": { }
}
---
<div>TRUE</div>

---
{
    "a": false,
    "b": { }
}
---
<div>FALSE</div>
