    <span class=["constant" value1]> "d" </span>
    <span class=["constant" value1 "two"]> "e" </span>

    <span class=[value1 "constant"]> "g" </span>
    <span class=value1 id="one"> "h" </span>  // has literal attribute, order will change
</div>
